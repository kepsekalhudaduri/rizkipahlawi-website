/**
 * Sanitizes and serializes database rows safely for the client.
 */
function serializeForClient(value) {
  if (Array.isArray(value)) {
    return value.map(function(item) {
      if (Array.isArray(item)) {
        return item.map(function(cell) {
          if (cell instanceof Date) { return cell.toISOString(); }
          if (typeof cell === 'undefined') { return null; }
          return cell;
        });
      } else if (typeof item === 'object' && item !== null) {
        const serializedObject = {};
        for (const key in item) {
          if (item.hasOwnProperty(key)) {
            const cell = item[key];
            if (cell instanceof Date) { serializedObject[key] = cell.toISOString(); }
            else if (typeof cell === 'undefined') { serializedObject[key] = null; }
            else { serializedObject[key] = cell; }
          }
        }
        return serializedObject;
      }
      return item;
    });
  } else if (typeof value === 'object' && value !== null) {
    const serializedObject = {};
    for (const key in value) {
      if (value.hasOwnProperty(key)) {
        const cell = value[key];
        if (cell instanceof Date) { serializedObject[key] = cell.toISOString(); }
        else if (typeof cell === 'undefined') { serializedObject[key] = null; }
        else { serializedObject[key] = cell; }
      }
    }
    return serializedObject;
  }
  return value;
}

/**
 * Retrieves all published blog posts from the spreadsheet.
 */
function getBlogPosts() {
  try {
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    const sheet = ss.getSheetByName('blog_posts');
    if (!sheet || sheet.getLastRow() <= 1) {
      return { ok: true, data: [] };
    }
    const data = sheet.getDataRange().getValues();
    const headers = data[0];
    const rows = data.slice(1);
    
    const blogPosts = [];
    for (let i = 0; i < rows.length; i++) {
      const row = rows[i];
      const post = {};
      headers.forEach(function(header, idx) {
        post[header] = row[idx];
      });
      if (post.status === 'Published') {
        blogPosts.push(post);
      }
    }
    
    // Reverse to show latest first
    blogPosts.reverse();
    return { ok: true, data: serializeForClient(blogPosts) };
  } catch (e) {
    return { ok: false, message: e.toString() };
  }
}

/**
 * Handles professional contact form submission.
 */
function submitContact(formData) {
  try {
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    const sheet = ss.getSheetByName('contacts');
    if (!sheet) {
      return { ok: false, message: 'Sheet contacts belum diinisialisasi.' };
    }
    
    sheet.appendRow([
      Utilities.getUuid(),
      formData.nama,
      formData.email,
      formData.pesan,
      new Date()
    ]);
    
    return { ok: true, message: 'Pesan Anda berhasil terkirim. Terima kasih atas kepercayaan Anda.' };
  } catch (e) {
    return { ok: false, message: e.toString() };
  }
}
